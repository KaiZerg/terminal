pwd - текущая папка
cd название_папки - сменить папку(change directory) 
cd .. - вернутся на уровень выше
ls - посмотреть содержимое папки
mkdir - создать новую папку
mv - переместить файл или папку
cp - копировать файл
touch - создать файл(+расширение)
rm - удаление файла
rmdir - удаление пустой папки
rm -R - удаление папки с файлами

cat - предпросмотр сожержимого файла | cat server.log
grep - фильтр, grep Error server.log
tail - выведи последние N строк, tail -5 server.log
head - выведи первые N строк, head -5 server.log
head -11 logs.txt | grep ERROR

vim или nano - открыть файл в редакторе

команда - man 
команда - --help
